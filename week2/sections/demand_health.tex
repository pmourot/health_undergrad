% Why important

% is the foundational theoretical framework of modern health economics.
% Before that, did not really have a model to understand why people invest in health and demand for health, why and how medical care is demanded,
% and how to think about the effect of income, education, aging on the demand for health.
% Health is not a weird exogenous thing that just happens to you.

% Health is not a good like others, it's something you invest in, more like a capital good
% WHen you consumed healthcare services, you don't consume heart surgery. That means its weird to have heart surgery enter your utility directly.
% What enters your utility is health, that you produce by investing in it with time and medical services

% Health is not entirely exogenous here: it's something you partially control by investing in it
% Caused quite a stir, negative edito in Financial World in 1972, upset to say that one chooses their health
% Health has two functions: gives you utility, but also allows you to work to consume

% The shadow price of health depends on many more things than medical care price (aging gamma, etc)
% Under some condition, we can have an increase of the shadow price that reduces health demanded but increases medical care demand (aging)
% We can also find that education has more effect on health than medicare care consumption ==> socio-economic determinants of health

% In the history of economics thought, this paper is a grandchild idea coming from Gary Becker (1965 and 1964).
% Gary Becker had indeed published a paper on HH production where what enters teh utility are commodities,
% produced by HH using goods and services bought on the market and their own time.
% In this model, goods are demanded not for themselves, but because they help produce commodities that enter the utility function.
% Also relevant is Becker's the theory of investment in human capital: investments in education raise future wage.

% Note: health and knowledge both enter the human capital in Grossman's model.
% However, he treats knowledge as more exogenous (that's how SES affects health investments: exogenous shift in MEC curve).
% Heckman's work has been on endogenizing knowledge.

% Who is Gary Becker? An economist who won the Nobel prize in economics in 1992 (noatbly for his work on human capital ==> investment in education and training)
% One of his big contributions is to have shown that microeconomic analysis can be applied to many aspects of human behavior
% including crime, health, mariage markets, family production, etc.
% He expanded to scope of economics.
% His most famous work: introduced and formalized the concept of human capital.
% showed it could be analyzed as a private investment decision with costs and returns — exactly like physical capital.

\begin{frame}{What is \textit{health}?}

    \begin{itemize}
        \item Something we enjoy: a \newbold{consumption good}
        \pause
        \item Something we invest in: a \newbold{stock/investment}
        \pause
        \item Something that affects our ability to produce: an \newbold{input} in
        \begin{itemize}
            \item The production of own health
            \item The production of productive time
        \end{itemize}

    \end{itemize}
\end{frame}
%---------------------------------------------------------------------
\begin{frame}{Single-period utility}

    Utility
    \begin{equation*}
        U = U\left( H_t, Z_t \right)
    \end{equation*}
    where
    \begin{itemize}
        \item $H_t$ is the level of health at time $t$
        \item $Z_t$ is a composite consumption good at time $t$
    \end{itemize}


\end{frame}
%---------------------------------------------------------------------
\begin{frame}{Single-period time constraint}

    An individual's activity:
    \begin{itemize}
        \item Spends time working (and earns a wage $w$ from it): $T^w$
        \item Spends time consuming the composite good $Z$: $T^Z$
        \item Spends time investing in health $H$: $T^H$
        \item If sick, needs to rest in bed and cannot do anything else: $T^s$
    \end{itemize}
    \vspace{3mm}

    Time constraint:
    \begin{equation*}
        \Theta = T^w + T^Z + T^H + T^s
    \end{equation*}
    where $\Theta$ is total available time in a period.

\end{frame}
%---------------------------------------------------------------------
\begin{frame}{Stock versus flow}

    \begin{equation*}
        U = U\left( H_t, Z_t \right)
    \end{equation*}

    Note:
    \begin{itemize}
        \item $Z_t$ is a \textit{flow}
        \item $H_t$ is a \newbold{stock}
    \end{itemize}
    $\implies$ How do you produce them?

\end{frame}
%---------------------------------------------------------------------
\begin{frame}{Production}

    Production functions:
    \begin{itemize}
        \item Health:
        \begin{equation*}
            H_t = H\left(H_{t-1}, T_t^H, M_t \right)
        \end{equation*}
        where
        \begin{itemize}
            \item $H_{t-1}$ is health stock from previous period
            \item $T_t^H$ is time invested in health this period
            \item $M_t$ are market inputs invested in health (e.g., vaccine, treadmill)
        \end{itemize}
        \pause
        \item Composite good:
        \begin{equation*}
            Z_t = Z\left( T_t^Z, J_t \right)
        \end{equation*}
        where
        \begin{itemize}
            \item $T_t^Z$ is time invested in consuming $Z$
            \item $J_t$ are market inputs invested in consuming $Z$ (e.g., food, video games)
        \end{itemize}
    \end{itemize}
\end{frame}
%---------------------------------------------------------------------
\begin{frame}{Production}

    Production functions:
    \begin{itemize}
        \item Health:
        \begin{equation*}
            H_t = H\left(H_{t-1}, T_t^H, M_t \right)
        \end{equation*}

        \item Composite good:
        \begin{equation*}
            Z_t = Z\left( T_t^Z, J_t \right)
        \end{equation*}
    \end{itemize}
    \vspace{5mm}
    $\implies$ Health is a \newbold{stock}: previous period's stock enters the production function\\
    $\implies$ You need time to get both consumption and health!
\end{frame}
%---------------------------------------------------------------------
\begin{frame}{Another constraint: the budget constraint}

    Time constraint \textit{(which we already saw)}:
    \begin{equation*}
        \Theta = T^w + T^Z + T^H + T^s
    \end{equation*}
    \pause

    Budget constraint:
    \begin{equation*}
        p_M M_t + p_J J_t \leq w T^w
    \end{equation*}
    where
    \begin{itemize}
        \item $p_M$ is the price of market inputs for health
        \item $p_J$ is the price of market inputs for consumption
        \item $w$ is the wage rate
    \end{itemize}

\end{frame}
%---------------------------------------------------------------------
\begin{frame}{Another constraint: the budget constraint}

    Time constraint \textit{(which we already saw)}:
    \begin{equation*}
        \Theta = T^w + T^Z + T^H + T^s
    \end{equation*}

    Budget constraint:
    \begin{equation*}
        p_M M_t + p_J J_t \textcolor{red}{=} w T^w
    \end{equation*}
    (no \textcolor{red}{saving/borrowing})\\

    \vspace{3mm}

    Note: the two constraints are \newbold{not independent}.

\end{frame}

\begin{frame}{Multi-period model}

    Assume multi-period utility as
    \begin{align*}
        U &= U(H_0, Z_0) + \delta U(H_1, Z_1) + \delta^2 U(H_2, Z_2) + \ldots + \delta^T U(H_T, Z_T)\\
          &=\sum_{t=0}^{T} \delta^t U\left( H_t, Z_t \right)
    \end{align*}
    where $\delta$ is the discount factor.
    \vspace{3mm}
    \pause

    Health as an investment good: introduce \textit{aging}
    \begin{equation*}
        H_t = H\left((1 - \gamma)H_{t-1}, T_t^H, M_t \right)
    \end{equation*}
    where $\gamma$ is the \newbold{depreciation rate} of health.

\end{frame}
